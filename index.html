function rgb(r, g, b){

  if(r == 0) { r = Number(0).toString(16) + 0}
  if(g == 0) { g = Number(0).toString(16)  + 0}
  if(b == 0) { b = Number(0).toString(16)  + 0}

  if(r > 255){r = 255}
  if(r < 0){r = 0}
  if(g > 255){g = 255}
  if(g < 0){g = 0}
  if(b > 255){b = 255}
  if(b < 0){b = 0}

  r = r.toString(16).toUpperCase()     
  g = g.toString(16).toUpperCase() 
  b = b.toString(16).toUpperCase()

  if (r.length < 2){ r = 0 + r}
  if (g.length < 2){ g = 0 + g}
  if (b.length < 2){ b = 0 + b}

  return r + g + b
};
